	RACIMOS_DIAS - CINTAS	 	%15
	CAJAS MERCADO NACIONAL		%16
	CLIENTES			%17
	MERCADO NACIONAL		%18
	CARGUE				%19
	CARGUE - MERCADO NACIONAL	%20
	PRODUCCIÓN			%21
	PRODUCCION - DÍAS		%22
	PRODUCCÓN - CAJAS PRODUCCIÓN	%23

	/* EMPRESAS */
	CREATE TABLE TblEmpresas(
		PKNit VARCHAR(11) PRIMARY KEY,
		Nombre VARCHAR(50) not null,
		Direccion VARCHAR(100) not null,
		Telefono VARCHAR(10) not null
	);
	
	/* FINCAS */
	CREATE TABLE TblFincas(
		PKIbm VARCHAR(5) PRIMARY KEY,
		Nombre VARCHAR(50) not null,
		Area_Neta FLOAT not null,
		Area_Bruta FLOAT not null,
		FKNit_TblEmpresas VARCHAR (11) not null,
		FOREIGN KEY (FKNit_TblEmpresas) REFERENCES TblEmpresas(PKNit)
	);

	/* LOTES */
	CREATE TABLE TblLotes(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		Lote VARCHAR(20) not null,
		Area_Neta FLOAT not null,
		Area_Bruta FLOAT not null,
		FKIbm_TblFincas VARCHAR(5) not null,
		FOREIGN KEY (FKIbm_TblFincas) REFERENCES TblFincas(PKIbm)	
	);

	/* TIPO FRUTA */
	CREATE TABLE TblTipoFruta(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		Descripcion VARCHAR(50) not null
	);

	/* CAJAS PRODUCCIÓN */
	CREATE TABLE TblCajasProduccion(
		PKCodigo VARCHAR(5) PRIMARY KEY,
		Descripcion VARCHAR(100) not null,
		FactorConversion FLOAT not null,
		FKId_TblTipoFruta INT not null,
		FOREIGN KEY (FKId_TblTipoFruta) REFERENCES TblTipoFruta(PKId)
	);

	/* CINTAS */
	CREATE TABLE TblCintas(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		Descripcion VARCHAR(50)
	);

	/* SEMANAS */
	CREATE TABLE TblSemanas(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		N_Semana VARCHAR(50),
		Fecha_Inicio DATE,
		Fecha_Fin DATE,
		Anho INT not null,
		FKId_TblCintas INT not null,
		FOREIGN KEY (FKId_TblCintas) REFERENCES TblCintas(PKId)
	);
	
	/* DÍAS */
	CREATE TABLE TblDias(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		Descripcion VARCHAR(50)
	);

	/* TIPO USUARIO */
	CREATE TABLE TblTipoUsuario(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		Descripcion VARCHAR(50)
	);

	/* ESTADO USUARIO */
	CREATE TABLE TblEstadoUsuario(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		Descripcion VARCHAR(50)
	);

	/* USUARIOS */
	CREATE TABLE TblUsuarios(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		FKIbm_TblFincas VARCHAR(5) not null,
		Password VARCHAR(255) not null,
		FKId_TblTipoUsuario INT not null,
		FKId_TblEstadoUsuario INT not null,
		FOREIGN KEY (FKIbm_TblFincas) REFERENCES TblFincas(PKIbm),
		FOREIGN KEY (FKId_TblTipoUsuario) REFERENCES TblTipoUsuario(PKId),
		FOREIGN KEY (FKId_TblEstadoUsuario) REFERENCES TblEstadoUsuario(PKId)	
	);

	/* EMBOLSE */
	CREATE TABLE TblEmbolse(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		FKId_TblSemanas INT not null,
		N_PlantasPresente INT not null,
		N_PlantasPrematuro INT not null,
		FOREIGN KEY (FKId_TblSemanas) REFERENCES TblSemanas(PKId)
	);

	/* RACIMOS */
	CREATE TABLE TblRacimos(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		FKId_TblSemanas INT not null,
		N_RacimosC INT not null,
		N_RacimosR INT not null,
		FOREIGN KEY (FKId_TblSemanas) REFERENCES TblSemanas(PKId)
	);
	
	/* DETALLE - RACIMOS - DIAS */
	CREATE TABLE TblDet_TblRacimos_TblDias(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		FKId_TblRacimos INT not null,
		FKId_TblDias INT not null,
		N_RacimosC_Dia INT not null,
		N_RacimosR_Dia INT not null,
		Total_PEmbarque INT not null,
		Total_POtrasFincas INT null,
		FOREIGN KEY (FKId_TblRacimos) REFERENCES TblRacimos(PKId),
		FOREIGN KEY (FKId_TblDias) REFERENCES TblDias(PKId)
	);
	
	/* DETALLE - DET_RACIMOS_DIAS - CINTAS */
	CREATE TABLE TblDet_TblDet_TblRacimos_TblDias(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		FKId_TblDet_TblRacimos_TblDias INT not null,
		FKId_TblCintas INT not null,
		N_RacimosC_Cintas INT not null,
		FOREIGN KEY (FKId_TblDet_TblRacimos_TblDias) REFERENCES TblDet_TblRacimos_TblDias(PKId),
		FOREIGN KEY (FKId_TblCintas) REFERENCES TblCintas(PKId)
	);

	/* MERCADO NACIONAL */
	CREATE TABLE TblCajasMercadoNacional(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		Descripcion VARCHAR(250) not null
	);
		
	CREATE TABLE TblMercadoNacional(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		Total_Elaborado INT not null
	);
	
	CREATE TABLE TblDet_TblMercadoNacional(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		FKId_TblMercadoNacional INT not null,
		FKId_TblDias INT not null,
		FKId_TblCajasMercadoNacional INT not null,
		Cantidad_Elaborado INT not null,
		FOREIGN KEY (FKId_TblMercadoNacional) REFERENCES TblMercadoNacional(PKId),
		FOREIGN KEY (FKId_TblDias) REFERENCES TblDias(PKId),
		FOREIGN KEY (FKId_TblCajasMercadoNacional) REFERENCES TblCajasMercadoNacional(PKId)
	);
	
	CREATE TABLE TblCargue(
		PKId_Cargue INTEGER PRIMARY KEY AUTO_INCREMENT,
		FKIbm_TblFincas VARCHAR(5) not null,
		Cliente VARCHAR(200) not null,
		Fecha_Cargue DATE not null,
		N_Poma VARCHAR(5) not null,
		DedoSuelto INT,
		Cluster INT,
		ManoEntera INT,
		Especial INT,
		Bolsa20Kilos INT,
		Bolsa25Kilos INT,
		Total INT,
		Placa_Vehiculo VARCHAR(8) not null,
		Conductor VARCHAR(200) not null,
		FOREIGN KEY (FKIbm_TblFincas) REFERENCES TblFincas(PKIbm)	
	);
	
	/* PRODUCCIÓN */
	CREATE TABLE TblProduccion(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		FKIbm_TblFincas VARCHAR(5) not null,
		FKId_TblEmbolse INT not null,
		FKId_TblRacimos INT not null,
		FKId_TblSemanas INT not null,
		FKId_TblMercadoNacional INT,
		Cod_Embarque VARCHAR(10) not null,
		Total_CElaboradas INT,
		Total_CREchazadas INT,
		Total_CExportadas INT,
		Ratio FLOAT,
		Merma FLOAT,
		Peso_Racimos FLOAT,
		Area_Recorrida FLOAT,
		Peso_Vastago FLOAT,
		Anho_Produccion VARCHAR(5) not null,
		FOREIGN KEY (FKIbm_TblFincas) REFERENCES TblFincas(PKIbm),
		FOREIGN KEY (FKId_TblEmbolse) REFERENCES TblEmbolse(PKId),
		FOREIGN KEY (FKId_TblRacimos) REFERENCES TblRacimos(PKId),
		FOREIGN KEY (FKId_TblSemanas) REFERENCES TblSemanas(PKId),
		FOREIGN KEY (FKId_TblMercadoNacional) REFERENCES TblMercadoNacional(PKId)
 	);

	/* DETALLE PRODUCCIÓN */
	CREATE TABLE TblDet_TblProduccion(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		FKId_TblProduccion INT not null,
		FKId_TblDias INT not null,
		Total_CE_Dia INT,
		Total_CR_Dia INT,
		Total_CEx_Dia INT,
		Ratio FLOAT,
		Merma FLOAT,
		Peso_Racimos FLOAT,
		Area_Recorrida FLOAT,
		Peso_Vastago FLOAT,
		Lotes_Cortados VARCHAR(50),
		Lotes_Iniciados VARCHAR(50),
		FOREIGN KEY (FKId_TblProduccion) REFERENCES TblProduccion(PKId),
		FOREIGN KEY (FKId_TblDias) REFERENCES TblDias(PKId)	
	);

	/* DETALLE - DETALLE PRODUCCION - CAJAS PRODUCCIÓN */
	CREATE TABLE TblDet_TblDet_TblProduccion(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		FKId_TblDet_TblProduccion INT not null,
		FKCodigo_TblCajasProduccion VARCHAR(5) not null,
		N_CajasProducidas_Dia INT,
		FOREIGN KEY (FKId_TblDet_TblProduccion) REFERENCES TblDet_TblProduccion(PKId),
		FOREIGN KEY (FKCodigo_TblCajasProduccion) REFERENCES TblCajasProduccion(PKCodigo)
	);

	/* EMBARQUE */
	CREATE TABLE TblEmbarque(
		PKCod VARCHAR(10) PRIMARY KEY,
    		FKId_TblSemanas INT not null,
    		Anho VARCHAR(5) not null,
    		FOREIGN KEY (FKId_TblSemanas) REFERENCES tblsemanas(PKId)
	);
	
	/* DETALLE EMBARQUE - FINCAS - CAJAS PRODUCCION */
	CREATE TABLE TblDet_TblEmbarque(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
    		FKCod_TblEmbarque VARCHAR(10) not null, 
    		FKIbm_TblFincas VARCHAR(5) not null,
    		FKCodigo_TblCajasProduccion VARCHAR(5) not null,
    		Cantidad INT not null,
    		FOREIGN KEY (FKCod_TblEmbarque) REFERENCES tblembarque(PKCod),
    		FOREIGN KEY (FKIbm_TblFincas) REFERENCES tblfincas(PKIbm),
    		FOREIGN KEY (FKCodigo_TblCajasProduccion) REFERENCES tblcajasproduccion(PKCodigo)
	);

	/* TABLAS INDEPENDIENTES DE CONFIGURACIÓN */
	CREATE TABLE TblRegistroSemanas(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
		Anho_generado INT not null
	);

	/* TABLA ESTIMATIVO */
	CREATE TABLE TblEstimativo(
		PKId INTEGER PRIMARY KEY AUTO_INCREMENT,
    		Finca VARCHAR(50) not null,
    		Premiun INT,
    		Especial INT,
    		FKCod_TblEmbarque VARCHAR(10) not null,
    		FOREIGN KEY (FKCod_TblEmbarque) REFERENCES TblEmbarque(PKCod)
	);
